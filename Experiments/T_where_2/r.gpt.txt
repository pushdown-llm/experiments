```sql
-- Create an index to speed up the filter condition
CREATE INDEX IF NOT EXISTS idx_orders_orderdate ON public.orders (o_orderdate);

-- Pushdown SQL to extract only rows in 1997
SELECT
  o_orderkey,
  o_custkey,
  o_orderstatus,
  o_totalprice,
  o_orderdate,
  o_orderpriority,
  o_clerk,
  o_shippriority,
  o_comment
FROM
  public.orders
WHERE
  o_orderdate >= DATE '1997-01-01'
  AND o_orderdate <= DATE '1997-12-31';
```

**Explanation:** Creates an index for `o_orderdate` to optimize the date filter scan.
