```sql
-- Create an index to speed up the join and filter
CREATE INDEX IF NOT EXISTS idx_customer_custkey_nationkey ON public.customer (c_custkey, c_nationkey);

-- Pushdown SQL query for the ETL transformation
SELECT 
  o.*,
  c.c_name,
  c.c_address,
  c.c_nationkey,
  c.c_phone,
  c.c_acctbal,
  c.c_mktsegment,
  c.c_comment
FROM 
  public.orders o
JOIN 
  public.customer c
  ON o.o_custkey = c.c_custkey
WHERE 
  c.c_nationkey = 10;
```

**Explanation:** The join and filter are pushed down with an index for efficient execution.
