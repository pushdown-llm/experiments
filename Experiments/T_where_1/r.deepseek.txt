-- Create an index to speed up the filter on o_orderpriority
CREATE INDEX IF NOT EXISTS idx_orders_orderpriority ON "public".orders(o_orderpriority);

-- Pushdown query with filter
SELECT
  o_orderkey,
  o_custkey,
  o_orderstatus,
  o_totalprice,
  o_orderdate,
  o_orderpriority,
  o_clerk,
  o_shippriority,
  o_comment
FROM "public".orders
WHERE o_orderpriority = '3-MEDIUM      ';
